    headA = headA.next
  while headB.next:
    headB = headB.next

  if headA == headB:
    return True

  return False

from heapq import heapify, heappop

def mergeL(lists):
  h = []

  for node in lists:
    while node:
      h.append(node.val)
      node = node.next

  if not h: return None

  heapify(h)

  root = ListNode(heappop(h))
  curNode = root

  while h:
    nextNode = ListNode(heappop(h))
    curNode.next = nextNode
    curNode = nextNode

  return root
